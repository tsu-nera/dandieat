{{/* This helper is used to generate a background color and text color from a string Used to generate deterministic colors for tags, categories, etc. Modified to work with older Hugo versions (< 0.154.0) */}} {{- $str := . -}} {{/* Use string length as a simple hash */}} {{- $len := len $str -}} {{- $hash := mul $len 17 -}} {{- $h := mod $hash 360 -}} {{- $backgroundColor := printf "hsl(%d, 60%%, 84%%)" $h -}} {{- $textColor := printf "hsl(%d, 60%%, 15%%)" $h -}} {{- return dict "BackgroundColor" $backgroundColor "TextColor" $textColor "Hue" $h -}}
